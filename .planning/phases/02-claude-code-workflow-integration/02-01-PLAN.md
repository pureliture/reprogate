#!/usr/bin/env python3
"""Tests for generate.py workflow command extension — AUTO-02.

Wave 0: These tests fail until plan 02-02 creates workflow template files and
plan 02-03 registers them in generate.py render_outputs().
"""
import pathlib
import sys
from typing import Any, Dict

import pytest
import yaml

sys.path.insert(0, str(pathlib.Path(__file__).resolve().parents[1]))

import generate  # noqa: E402


SCRIPTS_DIR = pathlib.Path(__file__).resolve().parents[1]
ROOT = SCRIPTS_DIR.parent
TEMPLATES_DIR = ROOT / "templates"


def _minimal_config(claude_enabled: bool = True) -> Dict[str, Any]:
    """Return a minimal config dict for generate tests."""
    return {
        "project": {"name": "test-project", "description": ""},
        "workspaces": {"primary": {"name": "primary", "branch": "main", "runtime": "python3"}},
        "processes": {"enabled": []},
        "tools": {"claude": {"enabled": claude_enabled, "hook_enforcement": False}, "codex": {"enabled": False}},
        "records": {
            "wp_path": "docs/work-packets",
            "adr_path": "records/adr",
            "changelog_path": "CHANGELOG.md",
        },
    }


class TestGenerateWorkflowCommands:
    """Tests for workflow command file generation via generate.render_outputs()."""

    def test_generate_produces_workflow_commands(self, tmp_path: pathlib.Path) -> None:
        """render_outputs with claude.enabled=True produces all three workflow-*.md files."""
        config = _minimal_config(claude_enabled=True)
        config_path = tmp_path / "reprogate.yaml"
        config_path.write_text(yaml.dump(config), encoding="utf-8")
        context = generate.context_from_config(config)

        produced = generate.render_outputs(
            output_root=tmp_path,
            config_path=config_path,
            config=config,
            context=context,
            force=True,
        )
        produced_names = [p.name for p in produced]

        assert "workflow-start.md" in produced_names, f"workflow-start.md not in {produced_names}"
        assert "workflow-status.md" in produced_names, f"workflow-status.md not in {produced_names}"
        assert "workflow-gate.md" in produced_names, f"workflow-gate.md not in {produced_names}"

    def test_commands_delegate_to_workflow_py(self, tmp_path: pathlib.Path) -> None:
        """Generated workflow-*.md files reference scripts/workflow.py (per D-02)."""
        config = _minimal_config(claude_enabled=True)
        config_path = tmp_path / "reprogate.yaml"
        config_path.write_text(yaml.dump(config), encoding="utf-8")
        context = generate.context_from_config(config)

        generate.render_outputs(
            output_root=tmp_path,
            config_path=config_path,
            config=config,
            context=context,
            force=True,
        )

        for cmd_name in ["workflow-start.md", "workflow-status.md", "workflow-gate.md"]:
            cmd_path = tmp_path / ".claude" / "commands" / cmd_name
            assert cmd_path.exists(), f"{cmd_name} not generated"
            content = cmd_path.read_text(encoding="utf-8")
            assert "scripts/workflow.py" in content, (
                f"{cmd_name} does not reference scripts/workflow.py — per D-02 all logic must be in scripts/"
            )

    def test_commands_do_not_contain_inline_logic(self, tmp_path: pathlib.Path) -> None:
        """Generated workflow-*.md files contain no Python def/import — no inline logic (D-02)."""
        config = _minimal_config(claude_enabled=True)
        config_path = tmp_path / "reprogate.yaml"
        config_path.write_text(yaml.dump(config), encoding="utf-8")
        context = generate.context_from_config(config)

        generate.render_outputs(
            output_root=tmp_path,
            config_path=config_path,
            config=config,
            context=context,
            force=True,
        )

        for cmd_name in ["workflow-start.md", "workflow-status.md", "workflow-gate.md"]:
            cmd_path = tmp_path / ".claude" / "commands" / cmd_name
            content = cmd_path.read_text(encoding="utf-8")
            assert "\ndef " not in content, f"{cmd_name} contains Python 'def' — logic belongs in workflow.py"
            assert "\nimport " not in content, f"{cmd_name} contains Python 'import' — logic belongs in workflow.py"

    def test_workflow_commands_not_generated_when_claude_disabled(self, tmp_path: pathlib.Path) -> None:
        """render_outputs with claude.enabled=False does not produce workflow command files."""
        config = _minimal_config(claude_enabled=False)
        config_path = tmp_path / "reprogate.yaml"
        config_path.write_text(yaml.dump(config), encoding="utf-8")
        context = generate.context_from_config(config)

        generate.render_outputs(
            output_root=tmp_path,
            config_path=config_path,
            config=config,
            context=context,
            force=True,
        )

        for cmd_name in ["workflow-start.md", "workflow-status.md", "workflow-gate.md"]:
            cmd_path = tmp_path / ".claude" / "commands" / cmd_name
            assert not cmd_path.exists(), f"{cmd_name} should not be generated when claude disabled"
