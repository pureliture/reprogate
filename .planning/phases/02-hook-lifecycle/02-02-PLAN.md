    ### session_start.py (HOOK-02)
    ```python
    #!/usr/bin/env python3
    """SessionStart hook — initializes current-session.json (HOOK-02)."""
    import datetime, json, os, pathlib, sys
    sys.path.insert(0, os.path.dirname(__file__))
    from reprogate_hook_base import check_disabled, get_profile

    ROOT = pathlib.Path(__file__).resolve().parents[2]
    SESSION_DATA = ROOT / ".claude" / "session-data"

    def main(session_data: pathlib.Path | None = None) -> int:
        check_disabled()
        data_dir = session_data or SESSION_DATA
        data_dir.mkdir(parents=True, exist_ok=True)
        now = datetime.datetime.utcnow().strftime("%Y%m%dT%H%M%SZ")
        payload = {
            "session_id": f"sess-{now}",
            "started_at": now,
            "profile": get_profile(),
            "tool_calls": [],
        }
        (data_dir / "current-session.json").write_text(
            json.dumps(payload, indent=2), encoding="utf-8"
        )
        return 0

    if __name__ == "__main__":
        sys.exit(main())
    ```

    ### session_stop.py (HOOK-03)
    ```python
    #!/usr/bin/env python3
    """Stop hook — saves session summary and observation YAML draft (HOOK-03)."""
    import datetime, json, os, pathlib, sys
    sys.path.insert(0, os.path.dirname(__file__))
    from reprogate_hook_base import check_disabled, get_profile

    ROOT = pathlib.Path(__file__).resolve().parents[2]
    SESSION_DATA = ROOT / ".claude" / "session-data"

    def main(session_data: pathlib.Path | None = None) -> int:
        check_disabled()
        data_dir = session_data or SESSION_DATA
        data_dir.mkdir(parents=True, exist_ok=True)
        now = datetime.datetime.utcnow().strftime("%Y%m%dT%H%M%SZ")
        profile = get_profile()

        # Load current session if exists
        current = {}
        current_path = data_dir / "current-session.json"
        if current_path.exists():
            try:
                current = json.loads(current_path.read_text(encoding="utf-8"))
            except Exception:
                current = {}

        session_id = current.get("session_id", f"sess-{now}")
        tool_calls = current.get("tool_calls", [])

        # Save summary JSON
        summary = {
            "session_id": session_id,
            "ended_at": now,
            "profile": profile,
            "tool_calls_count": len(tool_calls),
        }
        (data_dir / f"session-{now}-summary.json").write_text(
            json.dumps(summary, indent=2), encoding="utf-8"
        )

        # Save observation YAML draft
        obs_yaml = (
            f"observation_id: obs-{now}\n"
            f"session_id: {session_id}\n"
            f"captured_at: \"{now}\"\n"
            f"profile: {profile}\n"
            f"tool_calls_count: {len(tool_calls)}\n"
            "instincts: []\n"
            "notes: \"Auto-generated by Stop hook. Review and promote via /rg:learn-eval.\"\n"
        )
        (data_dir / f"session-{now}-observation.yaml").write_text(obs_yaml, encoding="utf-8")
        return 0

    if __name__ == "__main__":
        sys.exit(main())
    ```

    ### pretooluse_guard.py (HOOK-04 — PreCompact detection)
    The PreToolUse hook receives a JSON payload on stdin from Claude Code.
    This script must:
    1. Call check_disabled() — exits 0 (pass-through) if disabled
    2. Read stdin JSON
    3. If tool is Bash and command contains "compact": save pre-compact-state.json
    4. Always output allow decision (advisory only per HOOK-05 design)

    ```python
    #!/usr/bin/env python3
    """PreToolUse hook — PreCompact state capture (HOOK-04) + advisory governance (HOOK-05)."""
    import datetime, json, os, pathlib, sys
    sys.path.insert(0, os.path.dirname(__file__))
    from reprogate_hook_base import check_disabled, get_profile

    ROOT = pathlib.Path(__file__).resolve().parents[2]
    SESSION_DATA = ROOT / ".claude" / "session-data"

    def _read_payload() -> dict:
        try:
            raw = sys.stdin.read()
            return json.loads(raw) if raw.strip() else {}
        except Exception:
            return {}

    def _allow() -> None:
        print(json.dumps({"permissionDecision": "allow"}))

    def main(session_data: pathlib.Path | None = None) -> int:
        check_disabled()
        payload = _read_payload()
        data_dir = session_data or SESSION_DATA

        tool = payload.get("tool_name", "")
        command = (payload.get("tool_input") or {}).get("command", "")

        # HOOK-04: capture pre-compact state when compact is triggered
        if tool == "Bash" and "compact" in command.lower():
            data_dir.mkdir(parents=True, exist_ok=True)
            now = datetime.datetime.utcnow().strftime("%Y%m%dT%H%M%SZ")
            state = {
                "captured_at": now,
                "profile": get_profile(),
                "trigger": "compact",
                "session_data_snapshot": {},
            }
            (data_dir / "pre-compact-state.json").write_text(
                json.dumps(state, indent=2), encoding="utf-8"
            )

        _allow()
        return 0

    if __name__ == "__main__":
        sys.exit(main())
    ```
